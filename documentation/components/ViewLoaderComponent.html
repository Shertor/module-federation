import { Component, OnInit } from "@angular/core"

export type ViewLoaderStates = "show" | "hide"

  selector: "app-view-loader",
  templateUrl: "./view-loader.component.html",

  /** Состояние лоадера, 'show' - лоадер виден, 'hide' - лоадер не виден */

}
